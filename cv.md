/*RGB to HEX*/

```javascript
function rgb(r, g, b){
  return toHex(r)+toHex(g)+toHex(b);
}

function toHex(d) {

    if(d < 0 ) {
      return "00";
    }
    
    if(d > 255 ) {
      return "FF";
    }

    return ("0"+(Number(d).toString(16))).slice(-2).toUpperCase()
}
```
